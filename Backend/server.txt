var validateAdminUserToken = function(token) {
    jwt.verify(token, secretMessage, function(err, decoded){
        if (err) {
            return {result: false, message: err, login: null};
        }

        if(decoded.isAdmin) {
            return { result: true, message: null, login: decoded.login };
        } else {
            return {result: false, message: "Nie masz uprawnien do przegladania tej czesci strony", login: null };
        }
    });
}

app.post('/log-in', function (req, res) {
    console.log(JSON.stringify(req.body));
    UserModel.findOne({login: req.body.login}, function(err, user) {
        if (!user) {
            res.end( JSON.stringify({result: false, message: "Bledny login", token: null}));
        } else if (isValidPassword(user, req.body.password)) {
            const payload = {
                isAdmin: user.isAdmin,
                login: user.login
              };

            var token = jwt.sign(payload, secretMessage, {
                expiresIn: 300 // expires in 24 hours
            });

            res.end( JSON.stringify({result: true, message: null, user: 
                { 
                    login: user.login, 
                    id: user._id, 
                    token: token 
                }
            }) );
        } else {
            res.end( JSON.stringify({result: false, message: "Bledne haslo", token: null}));
        }
        console.log("Zakoczono zapytanie o logowanie");
    })
})

})

app.post('/sales', function (req, res) {
    const sale = req.body.sale;
    sales.push( { item: sale.item, nrOfPercentage: sale.nrOfPercentage, nrOfMinutes: sale.nrOfMinutes });
    setTimeout(notifyAboutSaleEnding(sale.item), sale.nrOfMinutes * 60000);

    io.sockets.send({ type: "sale-end", itemId: sale.item, nrOfPercentage: sale.nrOfPercentage });
})

var sales = [];
io.on('connection' , function(client) {
    console.log('Client connected...' );
});

notifyAboutSaleEnding = function(itemId) {
    sales = sales.filter(elem => elem.item !== itemId);
    io.sockets.send({type: "sale-end", itemId: itemId, nrOfPercentage: 0});
}
